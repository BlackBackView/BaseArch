# 协程缓存 — BaseArch Skill

## 功能概述
提供基于协程的缓存策略，支持内存缓存（LruCache）、请求缓存管理器，
以及可配置的缓存策略（缓存优先、网络优先等）。

package com.bbv.base.coroutine

import kotlinx.coroutines.*

/**
 * 协程作用域扩展
 */
fun <T> CoroutineScope.launchOnMain(block: suspend CoroutineScope.() -> T) {
    this.launch(Dispatchers.Main) { block() }
}

fun <T> CoroutineScope.launchOnIO(block: suspend CoroutineScope.() -> T): Job {
    return this.launch(Dispatchers.IO) { block() }
}

suspend fun <T> withIO(block: suspend CoroutineScope.() -> T): T {
    return withContext(Dispatchers.IO, block)
}

suspend fun <T> withDefault(block: suspend CoroutineScope.() -> T): T {
    return withContext(Dispatchers.Default, block)
}

### 3. MemoryCache.kt
```kotlin
package com.bbv.base.coroutine.cache

/**
 * 内存缓存实现
 */
class MemoryCache<K, V> : Cache<K, V> {

    private val map = HashMap<K, V>()

    override fun get(key: K): V? = map[key]

    override fun put(key: K, value: V) {
        map[key] = value
    }

    override fun remove(key: K) {
        map.remove(key)
    }

    override fun clear() {
        map.clear()
    }

    override val size: Int get() = map.size
}
```

### 4. LruCache.kt
```kotlin
package com.bbv.base.coroutine.cache

/**
 * LRU 缓存实现
 */
class LruCache<K, V>(private val maxSize: Int) : Cache<K, V> {

    private val cache = LinkedHashMap<K, V>(0, 0.75f, true)

    override fun get(key: K): V? = cache[key]

    override fun put(key: K, value: V) {
        cache[key] = value
        trimToSize()
    }

    override fun remove(key: K) {
        cache.remove(key)
    }

    override fun clear() {
        cache.clear()
    }

    override val size: Int get() = cache.size

    private fun trimToSize() {
        while (cache.size > maxSize) {
            val eldest = cache.entries.firstOrNull() ?: break
            cache.remove(eldest.key)
        }
    }
}
```

### 5. RequestCacheStrategy.kt
```kotlin
package com.bbv.base.coroutine.cache

/**
 * 缓存策略
 */
enum class RequestCacheStrategy {
    /** 仅网络 */
    ONLY_REMOTE,
    /** 仅缓存 */
    ONLY_CACHE,
    /** 缓存优先（缓存不存在则请求网络） */
    CACHE_FIRST,
    /** 网络优先（网络失败则使用缓存） */
    REMOTE_FIRST
}
```

### 6. RequestCacheManager.kt
```kotlin
package com.bbv.base.coroutine.cache

import kotlinx.coroutines.*

/**
 * 请求缓存管理器
 * 配合协程实现带缓存策略的数据请求
 */
class RequestCacheManager<K, V>(
    private val cache: Cache<K, V> = MemoryCache(),
    private val scope: CoroutineScope = CoroutineScope(Dispatchers.Main + SupervisorJob())
) {

    fun get(
        key: K,
        strategy: RequestCacheStrategy = RequestCacheStrategy.CACHE_FIRST,
        remoteFetch: suspend () -> V
    ): Deferred<V> = scope.async {
        when (strategy) {
            RequestCacheStrategy.ONLY_CACHE -> {
                cache.get(key) ?: throw IllegalStateException("Cache miss")
            }
            RequestCacheStrategy.ONLY_REMOTE -> {
                remoteFetch().also { cache.put(key, it) }
            }
            RequestCacheStrategy.CACHE_FIRST -> {
                cache.get(key) ?: remoteFetch().also { cache.put(key, it) }
            }
            RequestCacheStrategy.REMOTE_FIRST -> {
                try {
                    remoteFetch().also { cache.put(key, it) }
                } catch (e: Exception) {
                    cache.get(key) ?: throw e
                }
            }
        }
    }

    fun clear() {
        cache.clear()
    }

    fun remove(key: K) {
        cache.remove(key)
    }
}
```

## 使用方法
```kotlin
val cacheManager = RequestCacheManager<String, UserData>(LruCache(20))

// 缓存优先模式
cacheManager.get("user_123", RequestCacheStrategy.CACHE_FIRST) {
    // 网络请求
    api.fetchUserData("123")
}.onCompleted { userData ->
    updateUI(userData)
}

// 网络优先模式
cacheManager.get("user_123", RequestCacheStrategy.REMOTE_FIRST) {
    api.fetchUserData("123")
}
```
