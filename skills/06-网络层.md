# 网络层 — BaseArch Skill

## 功能概述
提供完整的网络请求封装，基于 Retrofit + OkHttp + Moshi，包含统一响应体解析、错误处理、请求拦截器、动态超时、JSON 适配器等。

### 3. BaseRetrofitServiceProvider.kt
```kotlin
package com.bbv.base.net

import okhttp3.OkHttpClient
import okhttp3.logging.HttpLoggingInterceptor
import retrofit2.Retrofit
import retrofit2.converter.moshi.MoshiConverterFactory
import java.util.concurrent.TimeUnit

open class BaseRetrofitServiceProvider {

    private val okHttpClient: OkHttpClient by lazy {
        OkHttpClient.Builder().apply {
            connectTimeout(NetConfig.connectTimeout, TimeUnit.SECONDS)
            readTimeout(NetConfig.readTimeout, TimeUnit.SECONDS)
            writeTimeout(NetConfig.writeTimeout, TimeUnit.SECONDS)
            if (NetConfig.isDebug) {
                addInterceptor(HttpLoggingInterceptor().apply {
                    level = HttpLoggingInterceptor.Level.BODY
                })
            }
            addInterceptor(CommonHeaderInterceptor())
        }.build()
    }

    private val retrofit: Retrofit by lazy {
        Retrofit.Builder()
            .baseUrl(NetConfig.baseUrl)
            .client(okHttpClient)
            .addConverterFactory(MoshiConverterFactory.create())
            .build()
    }

    protected fun <T> create(service: Class<T>): T = retrofit.create(service)
}
```

### 4. CommonHeaderInterceptor.kt
```kotlin
package com.bbv.base.net.interceptor

import okhttp3.Interceptor
import okhttp3.Response
import com.bbv.base.net.NetConfig

class CommonHeaderInterceptor : Interceptor {
    override fun intercept(chain: Interceptor.Chain): Response {
        val request = chain.request().newBuilder().apply {
            NetConfig.commonHeaders.forEach { (key, value) -> addHeader(key, value) }
            addHeader("Content-Type", "application/json")
            addHeader("Accept", "application/json")
        }.build()
        return chain.proceed(request)
    }
}
```

### 5. DynamicTimeoutInterceptor.kt
```kotlin
package com.bbv.base.net.interceptor

import okhttp3.Interceptor
import okhttp3.Response
import java.util.concurrent.TimeUnit

class DynamicTimeoutInterceptor : Interceptor {
    override fun intercept(chain: Interceptor.Chain): Response {
        val request = chain.request()
        val timeout = request.header("X-Timeout")?.toLongOrNull()
        return if (timeout != null) {
            chain.withConnectTimeout(timeout, TimeUnit.SECONDS)
                .withReadTimeout(timeout, TimeUnit.SECONDS)
                .withWriteTimeout(timeout, TimeUnit.SECONDS)
                .proceed(request.newBuilder().removeHeader("X-Timeout").build())
        } else {
            chain.proceed(request)
        }
    }
}
```

### 6. RetrofitException.kt
```kotlin
package com.bbv.base.net.exception

enum class ErrorKind {
    NETWORK, TIMEOUT, SERVER, PARSE, UNAUTHORIZED, UNKNOWN
}

class RetrofitException(
    val kind: ErrorKind,
    val code: Int = -1,
    override val message: String = ""
) : Exception(message) {
    companion object {
        fun networkError() = RetrofitException(ErrorKind.NETWORK, -1, "网络不可用")
        fun timeout() = RetrofitException(ErrorKind.TIMEOUT, -1, "请求超时")
        fun serverError() = RetrofitException(ErrorKind.SERVER, -1, "服务器异常")
        fun parseError() = RetrofitException(ErrorKind.PARSE, -1, "数据解析失败")
        fun unauthorized() = RetrofitException(ErrorKind.UNAUTHORIZED, 401, "未授权")
    }
}
```

## 使用方法
```kotlin
// 1. 初始化配置
NetConfig.baseUrl = "https://api.example.com/"
NetConfig.isDebug = BuildConfig.DEBUG
NetConfig.commonHeaders["Authorization"] = "Bearer $token"

// 2. 定义 API 接口
interface UserApi {
    @GET("user/list")
    suspend fun getUsers(@Query("page") page: Int): CommonResponse<List<User>>
}

// 3. 创建服务提供者
object UserServiceProvider : BaseRetrofitServiceProvider() {
    val api = create(UserApi::class.java)
}

// 4. 发起请求
viewModelScope.launch {
    val response = UserServiceProvider.api.getUsers(1)
    if (response.isSuccess()) {
        // 处理数据
    }
}
```
