### 1. Lg.kt — 日志工具
```kotlin
package com.bbv.base.util

import android.util.Log

/**
 * 日志工具类
 * Debug 模式输出全部日志，Release 模式关闭
 */
object Lg {
    private const val TAG = "BaseArch"
    var isDebug = false

    fun d(tag: String = TAG, msg: String) {
        if (isDebug) Log.d(tag, msg)
    }

    fun e(tag: String = TAG, msg: String, tr: Throwable? = null) {
        if (isDebug) {
            if (tr != null) Log.e(tag, msg, tr) else Log.e(tag, msg)
        }
    }

    fun i(tag: String = TAG, msg: String) {
        if (isDebug) Log.i(tag, msg)
    }

    fun json(tag: String = TAG, json: String) {
        if (isDebug) Log.d(tag, json)
    }
}
```

### 2. SpDelegate.kt — SharedPreferences 委托
```kotlin
package com.bbv.base.util

import android.content.Context
import android.content.SharedPreferences
import kotlin.properties.ReadWriteProperty
import kotlin.reflect.KProperty

/**
 * SharedPreferences 属性委托
 * 通过 by 委托简化 SP 读写
 */
class SpDelegate<T>(
    private val context: Context,
    private val name: String = "app_sp",
    private val default: T
) : ReadWriteProperty<Any?, T> {

    private val sp: SharedPreferences by lazy {
        context.getSharedPreferences(name, Context.MODE_PRIVATE)
    }

    @Suppress("UNCHECKED_CAST")
    override fun getValue(thisRef: Any?, property: KProperty<*>): T {
        return when (default) {
            is String -> sp.getString(property.name, default as String) as T
            is Int -> sp.getInt(property.name, default as Int) as T
            is Boolean -> sp.getBoolean(property.name, default as Boolean) as T
            is Long -> sp.getLong(property.name, default as Long) as T
            is Float -> sp.getFloat(property.name, default as Float) as T
            else -> default
        }
    }

    override fun setValue(thisRef: Any?, property: KProperty<*>, value: T) {
        sp.edit().apply {
            when (value) {
                is String -> putString(property.name, value)
                is Int -> putInt(property.name, value)
                is Boolean -> putBoolean(property.name, value)
                is Long -> putLong(property.name, value)
                is Float -> putFloat(property.name, value)
            }
        }.apply()
    }
}
```

### 3. SizeUtils.kt — 屏幕尺寸工具
```kotlin
package com.bbv.base.util

import android.content.res.Resources

object SizeUtils {
    private val metrics = Resources.getSystem().displayMetrics

    fun dp2px(dp: Float): Float = dp * metrics.density
    fun px2dp(px: Float): Float = px / metrics.density
    fun sp2px(sp: Float): Float = sp * metrics.scaledDensity
    fun px2sp(px: Float): Float = px / metrics.scaledDensity

    val screenWidth: Int get() = metrics.widthPixels
    val screenHeight: Int get() = metrics.heightPixels
}
```

### 5. DisplayKtx.kt — 显示相关扩展
```kotlin
package com.bbv.base.util

import android.app.Activity
import android.content.Context
import android.view.View
import android.view.inputmethod.InputMethodManager

fun Activity.hideKeyboard() {
    currentFocus?.let { view ->
        val imm = getSystemService(Context.INPUT_METHOD_SERVICE) as InputMethodManager
        imm.hideSoftInputFromWindow(view.windowToken, 0)
    }
}

fun View.showKeyboard() {
    val imm = context.getSystemService(Context.INPUT_METHOD_SERVICE) as InputMethodManager
    requestFocus()
    imm.showSoftInput(this, InputMethodManager.SHOW_IMPLICIT)
}
```

### 10. FileDownload.kt — 文件下载
```kotlin
package com.bbv.base.util

import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext
import java.io.File
import java.net.URL

object FileDownload {
    suspend fun download(url: String, destFile: File): File = withContext(Dispatchers.IO) {
        URL(url).openStream().use { input ->
            destFile.outputStream().use { output ->
                input.copyTo(output)
            }
        }
        destFile
    }
}
```
